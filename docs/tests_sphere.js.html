/* eslint-disable max-len */
const {Animation} = require('termination');
const BezierEasing = require('bezier-easing');

/**
 * @typedef SphereProperty
 * @property {number} y
 * @property {string} color
 * @property {string} label
 * @property {BezierEasing | string} func
 */

/**
 * @type {string}
 */
const sphereFrames = `\n  ______ \n .osmcns.\n.MNNMNMMN-\n.NNMNMMNM-\n .ohmdno.\n  ¯¯¯¯¯¯`;

/**
 * @type {SphereProperty[]}
 */
const sphereProperties = [
  {
    y: 0,
    color: 'red',
    label: 'cubic-bezier(0.25, -0.09, 0.82, 1.08)',
    // @ts-ignore
    func: (t) => new BezierEasing(0.25, -0.09, 0.82, 1.08)(t),
  },
  {
    y: 8,
    color: 'green',
    label: 'cubic-bezier(0.25, -0.09, 0, 0.98)',
    // @ts-ignore
    func: (t) => new BezierEasing(0.25, -0.09, 0, 0.98)(t),
  },
  {
    y: 16,
    color: 'yellow',
    label: 'ease-out',
    func: 'ease-out',
  },
  {
    y: 24,
    color: 'blue',
    label: 'linear',
    func: 'linear',
  },
];

/**
 * @type {Animation}
 */
const animation = new Animation({
  fps: 40,
  maxSize: {
    width: 100,
    height: 71,
  },
});

sphereProperties.map((sphereProperty) => {
  const sphereAnimation = animation.add({
    y: sphereProperty.y,
    x: 0,
    content: sphereProperty.label + '\n',
    color: sphereProperty.color,
    replaceSpace: true,
  });

  return sphereAnimation;
});

const spheres = sphereProperties.map((sphereProperty) => {
  const sphereAnimation = animation.add({
    y: sphereProperty.y,
    x: 0,
    color: sphereProperty.color,
    content: sphereFrames,
  });

  return sphereAnimation;
});

const spheresTransition = spheres.map((sphere, index) => sphere.transition([
  {
    props: {x: 65},
    duration: 1500,
    func: sphereProperties[index].func,
  },
], {loop: true, alternate: true, loopInterval: 400}));


animation.start();

console.log(spheresTransition);

spheresTransition.forEach((el) => el.run());

setTimeout(() => {
  animation.resume();
}, 5000);

setTimeout(() => {
  spheresTransition.forEach((el) => el.pause());
}, 20000);
